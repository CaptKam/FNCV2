<script>
    function switchTab(tabName) {
        // Reset all tabs
        document.querySelectorAll('.tab-btn').forEach(btn => {
            btn.classList.remove('active');
            btn.querySelector('.indicator').style.opacity = '0';
            btn.querySelector('.indicator').style.transform = 'scaleX(0)';
            btn.classList.remove('text-primary');
        });
        document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));
        
        // Activate selected
        const activeBtn = document.getElementById('tab-btn-' + tabName);
        activeBtn.classList.add('active');
        activeBtn.classList.add('text-primary');
        activeBtn.querySelector('.indicator').style.opacity = '1';
        activeBtn.querySelector('.indicator').style.transform = 'scaleX(1)';
        
        document.getElementById('tab-' + tabName).classList.add('active');
    }

    // Initialize state
    window.onload = () => {
        switchTab('ingredients');
    };
</script>
